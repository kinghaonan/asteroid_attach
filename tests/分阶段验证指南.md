## 分阶段验证

### 第一阶段：引力场学习验证

**验证内容**：
- PLY模型加载是否正确
- 多面体引力计算是否准确
- 神经网络模型训练是否正常

**运行示例**：

```python
from gravity_learning import PLYAsteroidModel, PolyhedralGravitySampler
from gravity_learning import GravityAndGradientDNN, GravityGradientTrainer

# 1. 加载PLY模型
model = PLYAsteroidModel("Castalia Radar-based.ply")
model.scale_to_real_size(target_diameter_m=1400.0)
print(f"小行星半径: {model.get_asteroid_radius()} m")

# 2. 生成引力样本
sampler = PolyhedralGravitySampler(model, asteroid_density=2670)
points = sampler.generate_sampling_points(num_samples=1000)
gravity = sampler.calculate_polyhedral_gravity(points)
print(f"引力计算完成: {gravity.shape}")

# 3. 训练DNN模型
dnn_model = GravityAndGradientDNN()
trainer = GravityGradientTrainer(dnn_model)
train_loader, val_loader = trainer.prepare_data(points, gravity, gradient)
trainer.train(train_loader, val_loader, epochs=50)

# 4. 验证模型
pred_gravity, pred_gradient = dnn_model.predict(points)
print(f"预测精度: {evaluate_accuracy(gravity, pred_gravity)}")
```

**成功标准**：
- [ ] PLY模型正常加载
- [ ] 引力计算无误
- [ ] DNN模型训练收敛
- [ ] 预测精度达到要求（<5%误差）

### 第二阶段：轨迹优化验证

**验证内容**：
- 各优化算法能否正常运行
- 优化结果是否合理
- 计算效率是否可接受

**运行示例**：

```python
from trajectory_optimization import ShootingMethodOptimizer, PseudospectralOptimizer
from trajectory_optimization import HomotopyOptimizer
import numpy as np

# 定义小行星和航天器（简化示例）
class Asteroid:
    def __init__(self):
        self.omega = np.array([0, 0, 2*np.pi/(5.27*3600)])
        self.mu = 4.463e5
    def gravity_gradient(self, r):
        r_norm = np.linalg.norm(r)
        return -self.mu * r / (r_norm**3) if r_norm > 1e-3 else np.zeros(3)
    def gravity_hessian(self, r):
        r_norm = np.linalg.norm(r)
        if r_norm < 1e-3:
            return np.zeros((3,3))
        I = np.eye(3)
        return -self.mu * (3*np.outer(r,r)/r_norm**5 - I/r_norm**3)

class Spacecraft:
    def __init__(self):
        self.T_max = 20.0
        self.I_sp = 400.0
        self.g0 = 9.81
        self.m0 = 1000.0

# 初始化
asteroid = Asteroid()
spacecraft = Spacecraft()

# 边界条件
r0 = np.array([10177, 6956, 8256])  # 初始位置 (m)
v0 = np.array([-25, -12, -17])       # 初始速度 (m/s)
m0 = spacecraft.m0                   # 初始质量 (kg)
rf = np.array([676, 5121, 449])      # 目标位置 (m)
vf = np.array([0, 0, 0])             # 目标速度 (m/s)
t_span = [0, 770]                    # 时间区间 (s)

# 1. 测试打靶法
print("=== 测试打靶法 ===")
shooting_opt = ShootingMethodOptimizer(asteroid, spacecraft)
result = shooting_opt.optimize_with_multiple_guesses(r0, v0, m0, rf, vf, t_span)
print(f"成功: {result['success']}, 燃料消耗: {m0 - result['final_mass']:.2f} kg")

# 2. 测试伪谱法
print("\n=== 测试伪谱法 ===")
ps_opt = PseudospectralOptimizer(asteroid, spacecraft, n_nodes=30)
result = ps_opt.optimize(r0, v0, m0, rf, vf, t_span)
print(f"成功: {result['success']}, 燃料消耗: {result['fuel_consumption']:.2f} kg")

# 3. 测试同伦法
print("\n=== 测试同伦法 ===")
if result['success']:
    homotopy_opt = HomotopyOptimizer(asteroid, spacecraft, n_steps=10)
    homotopy_result = homotopy_opt.solve_homotopy(
        r0, v0, m0, rf, vf, t_span,
        result['lam_r'][0], result['lam_v'][0], result['lam_m'][0]
    )
    print(f"燃料消耗: {homotopy_result['fuel_consumption']:.2f} kg")
```

**成功标准**：
- [ ] 打靶法正常收敛
- [ ] 伪谱法正常收敛
- [ ] 同伦法正常收敛
- [ ] 燃料消耗合理（<100 kg）
- [ ] 终端误差满足要求（位置<10m，速度<1m/s）

### 第三阶段：控制与仿真验证

**验证内容**：
- PID控制器能否跟踪轨迹
- 蒙特卡洛模拟是否正常
- 系统鲁棒性是否满足要求

**运行示例**：

```python
from control_simulation import PIDController, MonteCarloSimulator
from control_simulation import evaluate_project
import numpy as np

# 1. 测试PID控制器
print("=== 测试PID控制器 ===")
pid = PIDController(Kp=1.0, Ki=0.1, Kd=0.5)

# 模拟跟踪
ref_pos = np.array([1000, 500, 200])
ref_vel = np.array([10, 5, 2])
curr_pos = np.array([950, 480, 190])
curr_vel = np.array([9, 4.5, 1.8])

control = pid.compute_control(ref_pos, ref_vel, curr_pos, curr_vel, dt=0.1)
print(f"控制力: {control}")

# 2. 测试蒙特卡洛模拟
print("\n=== 测试蒙特卡洛模拟 ===")
from trajectory_optimization import ShootingMethodOptimizer

asteroid = Asteroid()
spacecraft = Spacecraft()
optimizer = ShootingMethodOptimizer(asteroid, spacecraft)

mc_simulator = MonteCarloSimulator(optimizer, asteroid, spacecraft, n_simulations=50)
stats = mc_simulator.run_monte_carlo(
    r0, v0, m0, rf, vf, t_span,
    position_noise=50.0,  # 50m位置扰动
    velocity_noise=2.0    # 2m/s速度扰动
)

print(f"成功率: {stats['success_rate']:.1f}%")
print(f"平均燃料消耗: {stats.get('fuel_mean', 0):.2f} kg")

# 3. 项目整体评估
print("\n=== 项目整体评估 ===")
evaluate_project()
```

**成功标准**：
- [ ] PID控制正常
- [ ] 蒙特卡洛模拟完成
- [ ] 成功率>80%
- [ ] 燃料消耗在合理范围内
